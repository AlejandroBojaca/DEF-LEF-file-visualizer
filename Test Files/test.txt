VERSION 5.8 ;
DIVIDERCHAR "/" ;
BUSBITCHARS "[]" ;
DESIGN MyCircuit ;
UNITS DISTANCE MICRONS 1000 ;

DIEAREA ( 0 0 ) ( 10000 10000 ) ;

// Components
COMPONENTS 3 ;
  - comp1 INV_X1
    + PLACED ( 1000 2000 ) N ;
  - comp2 AND2_X1
    + PLACED ( 4000 2000 ) N ;
  - comp3 DFF_X1
    + PLACED ( 7000 2000 ) N ;
END COMPONENTS

// Pins
PINS 3 ;
  - I1 + NET I1
    + DIRECTION INPUT
    + USE SIGNAL
    + LAYER METAL1 ( -100 0 ) ( 100 0 )
    + PLACED ( 1000 8000 ) N ;
  - O1 + NET O1
    + DIRECTION OUTPUT
    + USE SIGNAL
    + LAYER METAL1 ( -100 0 ) ( 100 0 )
    + PLACED ( 7000 8000 ) N ;
END PINS

// Nets
NETS 3 ;
  - I1 ;
  - net1 ;
  - O1 ;
END NETS

// Specialnets
SPECIALNETS 0 ;
END SPECIALNETS

END DESIGN
